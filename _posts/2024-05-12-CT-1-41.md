class Main {

    public int solution(int n, int k){
        int answer= 0;
        Queue<Integer> queue= new LinkedList<>();
        for(int i=1; i<n+1; i++){
            queue.offer(i);
        }
        while(queue.size() != 1){
            int count=1;
            for(int j=1; j<k; j++){
                count++;
                int num= queue.peek();
                queue.poll();
                queue.offer(num);
                if(count==k) queue.poll();
            }
        }
        answer= queue.peek();
        return answer;
    }
    public static void main(String[] args) {
        Main T = new Main();
        Scanner sc= new Scanner(System.in);
        int n= sc.nextInt();
        int k= sc.nextInt();
        System.out.print(T.solution(n, k));

    }
}
