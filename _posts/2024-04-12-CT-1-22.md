class Main {
    int[] dx= {-1, 0, 1, 0};
    int[] dy= {0, 1, 0, -1};
    public int solution(int n, int[][] intArr){
    int answer=0;
    for(int i=0; i<n; i++){
        for(int j=0; j<n; j++){
            boolean flag= true;
            for(int k=0; k<dx.length; k++){
                    int nx= i + dx[k];
                    int ny= j + dy[k];
                    if(nx>=0 && nx<n && ny>=0 && ny<n && intArr[i][j] <= intArr[nx][ny]) {
                        flag= false;
                        break;
                    }
            }
           if(flag) answer++;
        }
    }

    return answer;

}

            for(int j=0; j<n; j++){

        }
        System.out.print(T.solution(n, intArr));

}
