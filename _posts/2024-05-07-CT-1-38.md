class Main {

    public int solution(int n, int[][] board, int m, int[] moves){
        int answer=0;
        Stack<Integer> bucket= new Stack<>();
        for(int i=0; i<m; i++){
            int col= moves[i]-1;
            int row=0;
            for(int j=0; j<n; j++){
                if(board[j][col] !=0) {
                    row=j;
                    break;
                }
            }
            int doll= board[row][col];
            if(bucket.isEmpty()){
                bucket.push(doll);
            }else{
                if(bucket.peek() == doll){
                    bucket.pop();
                    answer++;
                }else{
                    bucket.push(doll);
                }
            }
            board[row][col] = 0;
        }
        answer= answer *2;
        return answer;
    }
    public static void main(String[] args) {
        Main T = new Main();
        Scanner sc= new Scanner(System.in);
        int n= sc.nextInt();
        int[][] board= new int[n][n];
        for(int i=0; i<n; i++){
            for(int j=0; j<n; j++){
                board[i][j] = sc.nextInt();
            }
        }
        int m= sc.nextInt();
        int[] moves= new int[m];
        for(int i=0; i<m; i++){
            moves[i]= sc.nextInt();
        }
        System.out.print(T.solution(n, board, m, moves));

    }
}
